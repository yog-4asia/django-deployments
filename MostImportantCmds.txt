#Create a virutal environment in 'Conda' and assign 'django' package to it
conda create --name <virtual-env-name> django
#activate 'django' virtual environment
activate myDjangoEnv
#Start a 'django' project 
django-admin startproject <project-name>
#Start a 'django' application within a 'django' project
python manage.py startapp <myapp>
#run the Web Server
python manage.py runserver
